@Override
public boolean onCreateOptionsMenu(Menu menu) {
    // 保存菜单
    mMenu = menu;
    // 清空菜单
    mMenu.clear();
    
    // 从XML提取菜单项
    MenuInflater inflater = getMenuInflater();
    inflater.inflate(R.menu.menu, menu);
    
    return true;
}
