方法一 - 操作画布

/**
 * 图片分割
 * @param g
 *            ：画布
 * @param paint
 *            ：画笔
 * @param imgBit
 *            ：图片
 * @param x
 *            ：X轴起点坐标
 * @param y
 *            ：Y轴起点坐标
 * @param w
 *            ：单一图片的宽度
 * @param h
 *            ：单一图片的高度
 * @param line
 *            ：第几列
 * @param row
 *            ：第几行
 */
public final void cuteImage(Canvas g, Paint paint, Bitmap imgBit, int x,
                            int y, int w, int h, int line, int row) {
    g.clipRect(x, y, x + w, h + y);
    g.drawBitmap(imgBit, x - line * w, y - row * h, paint);
    g.restore();
}
