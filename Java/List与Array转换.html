示例代码

// 声明数组类型和ArrayList类型对象
File[] files = new File[10];
List<File> listFiles = new ArrayList<File>();

// 填充List对象
if (files != null) {
    listFiles.addAll(Arrays.asList(files));
}

// 通过toArray()方法转换成数组
File[] files2 = listFiles.toArray(new File[] {});
