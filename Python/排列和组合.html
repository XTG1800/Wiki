组合

# 不含自身的组合
>>> for c in itertools.combinations([1, 2, 3, 4, 5], 3):
...     print ''.join(str(x) for x in c)

# 包含自身的组合
>>> for c in itertools.combinations_with_replacement([1, 2, 3, 4, 5], 3):
...     print ''.join(str(x) for x in c)

映射

>>> [x for x in itertools.product((1,2,3), (1,2))]
[(1, 1), (1, 2), (2, 1), (2, 2), (3, 1), (3, 2)]
