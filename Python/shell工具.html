命令行参数

import sh

ls = sh.ls.bake('-l')
ls('/home/zenki')

# 多个命令行参数需用,号隔开
sh.ls.('-l', '-m')

# 关键字参数
sh.ls.('-l', sort='time')

管道处理

import sh

sh.sort(sh.du(sh.glob('/home/zenki/*')), '-rn')

管道处理

import sh

# 加入参数：_piped=True
sh.sort(sh.du(sh.glob('/home/zenki/*'), _piped=True), '-rn')
